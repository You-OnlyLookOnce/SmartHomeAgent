## 💾 五、记忆系统

### 三级记忆结构

```python
class YueYueMemorySystem:
    """悦悦的记忆系统 - 分层管理"""
    
    # ========= 短期记忆（当前对话）=========
    short_term = {
        "current_context": "",           # 当前会话上下文
        "recent_messages": [],           # 最近 10 条消息
        "temporary_preferences": {},     # 临时偏好（本会话有效）
        "emotional_state": "neutral"     # 当前情感状态
    }
    
    # ========= 中期记忆（近期习惯）=========
    medium_term = {
        "daily_patterns": {},            # 每日作息规律
        "weekly_patterns": {},           # 每周活动规律
        "recent_feedback": [],           # 最近的用户反馈
        "pending_tasks": []              # 待办事项队列
    }
    
    # ========= 长期记忆（核心偏好）=========
    long_term = {
        "user_profile": {                # 用户画像
            "name": "",                  # 用户姓名
            "preferences": {             # 核心偏好
                "lighting": "warm_white",
                "temperature": 26,
                "music_taste": ["轻音乐", "民谣"],
                "sleep_time": "23:30",
                "wake_time": "07:30"
            },
            "avoid_topics": [],          # 不喜欢的话题
            "special_dates": []          # 重要日期（生日/纪念日等）
        },
        "interaction_history": [],       # 历史交互摘要（蒸馏后的）
        "learned_habits": []             # 学习到的生活习惯
    }
    
    async def update_memory(self, interaction: dict):
        """根据新交互更新各层记忆"""
        
        # 1. 添加到短期记忆
        self.short_term["recent_messages"].append(interaction)
        
        # 2. 定期总结到中期记忆
        if len(self.short_term["recent_messages"]) >= 20:
            summary = await self.summarize_recent()
            self.medium_term["daily_patterns"].update(summary)
            self.short_term["recent_messages"] = []
        
        # 3. 夜间蒸馏到长期记忆（每天凌晨 2 点）
        # （由定时任务触发）
    
    def get_relevant_memory(self, query_type: str) -> dict:
        """获取相关记忆"""
        
        relevant = {}
        
        # 从各层检索
        if query_type in ["context", "immediate"]:
            relevant["short_term"] = self.short_term
        
        if query_type in ["patterns", "habits"]:
            relevant["medium_term"] = self.medium_term
        
        if query_type in ["preferences", "profile"]:
            relevant["long_term"] = self.long_term
        
        return relevant
```
